173 void Domotification::gererNotifications(QVector<bool> etatsmachines,

177  qDebug() << Q_FUNC_INFO << "etatsmachines" << etatsmachines << "poubelles" << etatspoubelles
178  << "etatboite" << etatboite;

180  {
181  if(etatspoubelles[i])

183  switch(i)

185  case 0:
186  getPoubelles()[0]->setNotifie(true);
187  notifier(NOTIFICATION_POUBELLE_BLEUE);
188  break;
189  case 1:
190  getPoubelles()[1]->setNotifie(true);
191  notifier(NOTIFICATION_POUBELLE_VERTE);
192  break;
193  case 2:
194  getPoubelles()[2]->setNotifie(true);
195  notifier(NOTIFICATION_POUBELLE_JAUNE);
196  break;
197  case 3:
198  getPoubelles()[3]->setNotifie(true);
199  notifier(NOTIFICATION_POUBELLE_ROUGE);
200  break;
201  case 4:
202  getPoubelles()[4]->setNotifie(true);
203  notifier(NOTIFICATION_POUBELLE_GRISE);
204  break;
205  default:
206  break;
207  }
208  }
209  }
210  for(auto i = 0; i < etatsmachines.size(); i++)
211  {
212  if(etatsmachines[i])
213  {
214  getMachines()[i]->setNotifie(true);
215  notifier(NOTIFICATION_MACHINE);
216  qDebug() << Q_FUNC_INFO << "etatNotificationMachines" << getMachines()[i]->estNotifie();
217  getNotificationModule("machine", i);
218  }
219  }
220  if(etatboite)
221  {
222  getBoite()->setNotifie(true);
223  notifier(NOTIFICATION_BOITE);
224  }

405  parametres.setValue(nomSection + "/nom", machines[i]->getNom());

409  }

412  if(boite == nullptr)

422  }
